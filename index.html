app.delete('/api/produtos/:id', async (req, res) => { try { const apagado = await Produto.findByIdAndDelete(req.params.id); if (!apagado) return res.status(404).json({ mensagem: "Produto não encontrado" }); res.json({ mensagem: "Removido" }); } catch (err) { res.status(500).json({ mensagem: "Erro ao apagar produto" }); } });
